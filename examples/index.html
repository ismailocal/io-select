AJAX Data Loading

These examples use a client-side mock of $.ajax to simulate server interactions. Open your browser's console to see mock AJAX call logs. 

$('#ajaxSelect').ioSelect({
    placeholder: 'Search remote data (AJAX)...',
    ajax: {
        url: '/api/options/mock',
        delay: 400,
        triggerMinSearchTermLength: 0
    }
});
Selected Value: 
null
$('#ajaxSelectWithInitial').ioSelect({
    placeholder: 'AJAX with Pre-selected Values...',
    ajax: {
        url: '/api/options/mock',
        initialValueUrl: '/api/options/initial-mock',
        delay: 400,
        triggerMinSearchTermLength: 1
    }
});

[]
